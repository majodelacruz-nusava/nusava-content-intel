Bridging the Last 10%

The alternative stack can't access data locked inside your TikTok accounts. Here's what's missing and how to get it with a 15-minute weekly export routine.

Missing Data Source How to Get It Time
Exact order-level GMV TikTok Seller Center Orders tab → Export CSV weekly → Upload to database 3 min
Video-to-sale attribution Seller Center → Content Content tab shows GMV per video → Export CSV → Match to video IDs 3 min
Audience demographics TikTok Analytics (in-app) Analytics → Followers tab → Screenshot or manual entry weekly 3 min
Watch-through / retention TikTok Studio Video detail → Retention graph → Screenshot for AI analysis 3 min
LIVE real-time metrics During LIVE session Assign team member to log peak viewers, items sold, key moments in a Google Sheet During LIVE

Suggested Weekly Workflow

Step 1: Export (5 min)

Log into Seller Center → Export Orders CSV + Content Performance CSV. Save to a shared Google Drive folder.

Step 2: Demographics (3 min)

Open TikTok Analytics → Screenshot follower demographics → Drop into the same Drive folder.

Step 3: Ingest (automatic)

Ingestion script watches the Drive folder → Parses CSVs → Updates database → AI re-analyzes with fresh data.

This can be automated further with Google Sheets API + Apps Script to auto-format exports, or with Zapier/Make to trigger ingestion on file upload.

import { ApifyClient } from 'apify-client';

const client = new ApifyClient({ token: process.env.APIFY_API_TOKEN });

export async function scrapeNusavaVideos() {
  const run = await client.actor('clockworks/tiktok-scraper').call({

  const { items } = await client.dataset(run.defaultDatasetId).listItems();
  return items;
}

export async function detectPersona(videoData: any) {

  });
  return JSON.parse(response.content[0].text);
}

# Automated CSV ingestion (scripts/ingest-competitor-data.ts):
import { parse } from 'csv-parse/sync';
import fs from 'fs';
import { supabase } from '../lib/supabase';

export async function ingestCompetitorCSV(filePath: string) {
  const content = fs.readFileSync(filePath, 'utf-8');
  const records = parse(content, { columns: true, skip_empty_lines: true });

  for (const record of records) {
    await supabase.from('competitor_products').upsert({
      product_name: record['Product Name'],
      estimated_sales: parseInt(record['Est. Sales']),
      estimated_revenue: parseFloat(record['Est. Revenue']),
      top_creator: record['Top Creator'],
      video_count: parseInt(record['Videos']),
      updated_at: new Date().toISOString(),
    });
  }
  console.log(`Ingested ${records.length} competitor records`);
}

-- Products table
CREATE TABLE products (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  tiktok_product_id TEXT UNIQUE,
  name TEXT NOT NULL,
  price DECIMAL(10,2),
  category TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Videos table
CREATE TABLE videos (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  tiktok_video_id TEXT UNIQUE,
  product_id UUID REFERENCES products(id),
  title TEXT,
  creator TEXT,
  views INTEGER DEFAULT 0,
  likes INTEGER DEFAULT 0,
  shares INTEGER DEFAULT 0,
  comments INTEGER DEFAULT 0,
  ctr DECIMAL(5,2),
  conversion_rate DECIMAL(5,2),
  gmv DECIMAL(10,2) DEFAULT 0,
  persona TEXT,
  persona_confidence DECIMAL(3,2),
  hook_text TEXT,
  scraped_at TIMESTAMPTZ DEFAULT now()
);

-- LIVE sessions table
CREATE TABLE live_sessions (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  title TEXT,
  date TIMESTAMPTZ,
  duration_minutes INTEGER,
  peak_viewers INTEGER,
  avg_viewers INTEGER,
  comments INTEGER,
  items_sold INTEGER,
  gmv DECIMAL(10,2) DEFAULT 0,
  scraped_at TIMESTAMPTZ DEFAULT now()
);

-- Content briefs table
CREATE TABLE content_briefs (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  brief_type TEXT, -- 'hooks', 'scripts', 'shooting_guide', 'live_playbook'
  content JSONB,
  generated_at TIMESTAMPTZ DEFAULT now()
);

-- Persona analytics
CREATE TABLE persona_stats (
  persona TEXT PRIMARY KEY,
  video_count INTEGER DEFAULT 0,
  total_views BIGINT DEFAULT 0,
  total_gmv DECIMAL(12,2) DEFAULT 0,
  avg_engagement DECIMAL(5,2),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Competitor data
CREATE TABLE competitor_products (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  product_name TEXT,
  estimated_sales INTEGER,
  estimated_revenue DECIMAL(10,2),
  top_creator TEXT,
  video_count INTEGER,
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Create indexes for fast queries
CREATE INDEX idx_videos_product ON videos(product_id);

# lib/supabase.ts — Database client:
import { createClient } from '@supabase/supabase-js';

  process.env.SUPABASE_URL!,

// Example: Get top videos by GMV
export async function getTopVideos(limit = 15) {
  const { data } = await supabase
    .from('videos')
    .select('*, products(name)')
    .order('gmv', { ascending: false })
    .limit(limit);

}
